# Домашнее задание к работе 9

## Условие задачи
Написать программу, выводящую на экран заданную геометрическую фигуру (квадрат), 
нарисованную с помощью заданного с клавиатуры символа. Предусмотреть возможность изменения размеров фигуры путем задания ее размеров в количестве символов пользователем.

## 1. Алгоритм и блок-схема

### Алгоритм
1. **Начало**
2. Объявить переменных:
   - `size` - размер стороны квадрата.
   - `symbol` - символ для рисования.
   - `choice` - выбор пользователя о продолжении.
3. Цикл выполнения (do-while)
4. **Конец**

### Блок-схема 




## 2. Реализация программы
#include <stdio.h>
#include <locale.h>

int main() 
{
    setlocale(LC_ALL, "RUS");
    int size;
    char symbol;
    char choice;

    do {
        printf("Введите размер стороны квадрата: ");
        scanf("%d", &size);
        printf("Введите символ: ");
        scanf(" %c", &symbol);

        printf("\n");

        for (int i = 0; i < size; i++) {
            for (int j = 0; j < size; j++) {
                if (i == 0  i == size - 1  j == 0 || j == size - 1) {
                    printf("%c ", symbol);
                } else {
                    printf("  ");
                }
            }
            printf("\n");
        }
        printf("\nХотите продолжить? (Да - введите 1, Нет - введите 0): ");
        scanf(" %c", &choice);
        getchar();

    } while (choice == '1' || choice == '1');

    printf("Программа завершена.\n");
    return 0;
}
